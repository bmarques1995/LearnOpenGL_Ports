cmake_minimum_required(VERSION 3.20)

project(ShadersInterpolation LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)

add_subdirectory(./vendor/glfw)
add_subdirectory(./vendor/glad)

add_executable(ShadersInterpolation src/entry.cc)
target_include_directories(ShadersInterpolation PRIVATE ./vendor/glfw/include ./vendor/glad/include)
target_link_libraries(ShadersInterpolation PRIVATE glfw glad)
